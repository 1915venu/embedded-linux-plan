DAYS 38-39: THREAD POOL IMPLEMENTATION

🧠 THEORY: Why Thread Pools?

WITHOUT Thread Pool:
  Task 1 → Create Thread → Execute → Destroy Thread  (expensive!)
  Task 2 → Create Thread → Execute → Destroy Thread  (syscall + stack alloc!)
  Task 3 → Create Thread → Execute → Destroy Thread  (repeat!)

WITH Thread Pool:
  Init: Create N workers → All wait for tasks
  Task 1 → Wake Worker → Execute → Worker waits again  (just a wake + context switch)
  Task 2 → Wake Worker → Execute → Worker waits again  (reuse existing thread!)

Thread creation cost: ~50μs (clone syscall, stack allocation, scheduling)
Task dispatch in pool: ~1μs (signal condition variable, context switch)
→ 50x improvement for short tasks!

 * - Producer-consumer pattern
 * - Clean API design
 * - Resource management and graceful shutdown
 */

/**
 * Create a thread pool with @num_threads worker threads.
 * Returns NULL on failure.
 */
thread_pool_t *thread_pool_create(size_t num_threads);

/**
 * Submit a task for execution.
 * @pool: the thread pool
 * @func: function to execute
 * @arg:  argument passed to func
 * Returns 0 on success, -1 if pool is shutting down.
 */
int thread_pool_submit(thread_pool_t *pool, task_func_t func, void *arg);

/**
 * Wait for all submitted tasks to complete.
 * Blocks until queue is empty AND all tasks finish.
 */
void thread_pool_wait(thread_pool_t *pool);

/**
 * Destroy the pool. Waits for current tasks, then joins threads.
 * After this call, pool pointer is invalid.
 */
void thread_pool_destroy(thread_pool_t *pool);

/**
 * Get pool statistics.
 */
size_t thread_pool_pending(thread_pool_t *pool);
size_t thread_pool_active(thread_pool_t *pool);

#endif /* THREAD_POOL_H */

💻 COMPLETE CODE: Thread Pool Implementation

// File: ~/embedded_learning/day_38/thread_pool/thread_pool.c
#include "thread_pool.h"
#include <pthread.h>
#include <stdlib.h>
#include <stdio.h>
#include <string.h>

/* ==== TASK QUEUE NODE ==== */
typedef struct task_node {
    task_func_t func;
    void *arg;
    struct task_node *next;
} task_node_t;

/* ==== THREAD POOL STRUCTURE ==== */
struct thread_pool {
    pthread_t *threads;         /* Worker thread array */
    size_t num_threads;         /* Number of workers */

    task_node_t *queue_head;    /* Task queue (linked list) */
    task_node_t *queue_tail;
    size_t queue_size;          /* Tasks in queue */

    size_t active_tasks;        /* Tasks currently executing */

    pthread_mutex_t lock;       /* Protects all shared state */
    pthread_cond_t not_empty;   /* Signal: task available */
    pthread_cond_t all_done;    /* Signal: all tasks complete */

    bool shutdown;              /* Shutdown flag */
};

/* ==== WORKER THREAD FUNCTION ==== */
static void *worker_func(void *arg)
{
    thread_pool_t *pool = (thread_pool_t *)arg;

    while (1) {
        task_func_t func;
        void *task_arg;

        /* ==== CRITICAL SECTION: Get task from queue ==== */
        pthread_mutex_lock(&pool->lock);

        /* Wait for task or shutdown */
        /* WHY while() not if(): Spurious wakeups + broadcast
         * After broadcast, multiple threads wake but only one
         * should get each task. while re-checks condition. */
        while (pool->queue_size == 0 && !pool->shutdown) {
            pthread_cond_wait(&pool->not_empty, &pool->lock);
        }

        /* Check shutdown AFTER waking */
        if (pool->shutdown && pool->queue_size == 0) {
            pthread_mutex_unlock(&pool->lock);
            break;  /* Exit: shutdown + empty queue */
        }

        /* Dequeue task */
        task_node_t *node = pool->queue_head;
        func = node->func;
        task_arg = node->arg;
        pool->queue_head = node->next;
        if (!pool->queue_head)
            pool->queue_tail = NULL;
        pool->queue_size--;
        pool->active_tasks++;
        free(node);

        pthread_mutex_unlock(&pool->lock);
        /* ==== END CRITICAL SECTION ==== */

        /* Execute task (outside lock — don't block other workers!) */
        func(task_arg);

        /* Task complete — update counter */
        pthread_mutex_lock(&pool->lock);
        pool->active_tasks--;

        /* If all done: signal thread_pool_wait() */
        if (pool->queue_size == 0 && pool->active_tasks == 0) {
            pthread_cond_signal(&pool->all_done);
        }
        pthread_mutex_unlock(&pool->lock);
    }

    return NULL;
}

/* ==== PUBLIC API ==== */

thread_pool_t *thread_pool_create(size_t num_threads)
{
    if (num_threads == 0)
        return NULL;

    thread_pool_t *pool = calloc(1, sizeof(thread_pool_t));
    if (!pool)
        return NULL;

    pool->num_threads = num_threads;
    pool->threads = calloc(num_threads, sizeof(pthread_t));
    if (!pool->threads) {
        free(pool);
        return NULL;
    }

    pthread_mutex_init(&pool->lock, NULL);
    pthread_cond_init(&pool->not_empty, NULL);
    pthread_cond_init(&pool->all_done, NULL);

    /* Create worker threads */
    for (size_t i = 0; i < num_threads; i++) {
        if (pthread_create(&pool->threads[i], NULL, worker_func, pool) != 0) {
            /* Cleanup on failure */
            pool->shutdown = true;
            pthread_cond_broadcast(&pool->not_empty);
            for (size_t j = 0; j < i; j++)
                pthread_join(pool->threads[j], NULL);
            free(pool->threads);
            free(pool);
            return NULL;
        }
    }

    return pool;
}

int thread_pool_submit(thread_pool_t *pool, task_func_t func, void *arg)
{
    if (!pool || !func)
        return -1;

    task_node_t *node = malloc(sizeof(task_node_t));
    if (!node)
        return -1;

    node->func = func;
    node->arg = arg;
    node->next = NULL;

    pthread_mutex_lock(&pool->lock);

    if (pool->shutdown) {
        pthread_mutex_unlock(&pool->lock);
        free(node);
        return -1;
    }

    /* Enqueue */
    if (pool->queue_tail) {
        pool->queue_tail->next = node;
    } else {
        pool->queue_head = node;
    }
    pool->queue_tail = node;
    pool->queue_size++;

    /* Wake one worker */
    pthread_cond_signal(&pool->not_empty);

    pthread_mutex_unlock(&pool->lock);
    return 0;
}

void thread_pool_wait(thread_pool_t *pool)
{
    if (!pool)
        return;

    pthread_mutex_lock(&pool->lock);
    while (pool->queue_size > 0 || pool->active_tasks > 0) {
        pthread_cond_wait(&pool->all_done, &pool->lock);
    }
    pthread_mutex_unlock(&pool->lock);
}

void thread_pool_destroy(thread_pool_t *pool)
{
    if (!pool)
        return;

    /* Set shutdown flag and wake all workers */
    pthread_mutex_lock(&pool->lock);
    pool->shutdown = true;
    pthread_cond_broadcast(&pool->not_empty);  /* Wake ALL workers */
    pthread_mutex_unlock(&pool->lock);

    /* Join all threads (wait for them to finish) */
    for (size_t i = 0; i < pool->num_threads; i++) {
        pthread_join(pool->threads[i], NULL);
    }

    /* Free remaining queued tasks */
    task_node_t *node = pool->queue_head;
    while (node) {
        task_node_t *next = node->next;
        free(node);
        node = next;
    }

    /* Cleanup */
    pthread_mutex_destroy(&pool->lock);
    pthread_cond_destroy(&pool->not_empty);
    pthread_cond_destroy(&pool->all_done);
    free(pool->threads);
    free(pool);
}

size_t thread_pool_pending(thread_pool_t *pool)
{
    pthread_mutex_lock(&pool->lock);
    size_t n = pool->queue_size;
    pthread_mutex_unlock(&pool->lock);
    return n;
}

size_t thread_pool_active(thread_pool_t *pool)
{
    pthread_mutex_lock(&pool->lock);
    size_t n = pool->active_tasks;
    pthread_mutex_unlock(&pool->lock);
    return n;
}

// File: ~/embedded_learning/day_38/thread_pool/test_app.c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <assert.h>
#include <pthread.h>
#include <time.h>
#include "thread_pool.h"

static pthread_mutex_t counter_lock = PTHREAD_MUTEX_INITIALIZER;
static int task_counter = 0;

void simple_task(void *arg) {
    int id = *(int*)arg;
    usleep(10000 + (rand() % 50000));

    pthread_mutex_lock(&counter_lock);
    task_counter++;
    pthread_mutex_unlock(&counter_lock);

    printf("  Task %d completed (thread %lu)\n", id, pthread_self());
}

void stress_task(void *arg) {
    pthread_mutex_lock(&counter_lock);
    task_counter++;
    pthread_mutex_unlock(&counter_lock);
}

int main(void) {
    int passed = 0, failed = 0;

    printf("=== Thread Pool Test Suite ===\n\n");

    /* Test 1: Basic creation and destruction */
    printf("--- Test 1: Create/Destroy ---\n");
    {
        thread_pool_t *pool = thread_pool_create(4);
        assert(pool != NULL);
        thread_pool_destroy(pool);
        passed++;
        printf("[PASS] Create and destroy\n\n");
    }

    /* Test 2: Submit and execute tasks */
    printf("--- Test 2: Submit Tasks ---\n");
    {
        thread_pool_t *pool = thread_pool_create(4);
        task_counter = 0;

            thread_pool_submit(pool, simple_task, &ids[i]);

        thread_pool_wait(pool);
        thread_pool_destroy(pool);

        if (task_counter == 10) {
            printf("[PASS] All 10 tasks executed (counter=%d)\n\n", task_counter);
            passed++;
        } else {
            printf("[FAIL] Expected 10, got %d\n\n", task_counter);
            failed++;
        }
    }

    /* Test 3: Stress test (many tasks) */

        thread_pool_wait(pool);
        thread_pool_destroy(pool);

        if (task_counter == 1000) {
            printf("[PASS] 1000 tasks completed correctly\n\n");

            printf("[FAIL] Expected 1000, got %d\n\n", task_counter);

📝 INTERVIEW QUESTIONS — Thread Pool (10 Questions)

Q1: Explain your thread pool architecture.

Producer-consumer pattern. N workers created at init, block on condition variable. Main thread submits tasks to queue (mutex-protected). Signal wakes one worker, which pops task and executes outside lock. Two condition vars: not_empty for workers, all_done for pool_wait().

Q2: Why while loop instead of if for cond_wait?

Spurious wakeups (POSIX allows). Also, broadcast wakes multiple threads but only one should get each task. while re-checks condition after each wakeup. Without while: could dequeue from empty queue → crash.

Q3: How do you handle graceful shutdown?

Set shutdown flag, broadcast not_empty to wake ALL workers. Each worker checks shutdown && queue_empty → exits. Join all threads. Free remaining queue nodes and pool memory. All pending tasks complete before shutdown.

Q4: Why signal vs broadcast?

Signal: wakes ONE waiter. Use for submit (one task, one worker). Broadcast: wakes ALL waiters. Use for shutdown (all workers must check flag). Using broadcast for submit wastes CPU (thundering herd).

Q5: How does thread_pool_wait work?

Checks queue_size == 0 && active_tasks == 0. If not, waits on all_done condition. Workers signal all_done after completing a task when both conditions are true. Uses while loop for spurious wakeups.

Q6: Why execute tasks outside the lock?

Lock protects queue access, not task execution. Holding lock during execution would serialize ALL tasks — defeating the purpose of threading. Pop task with lock, execute without lock, re-lock to update active_tasks.

Q7: How do you make the pool thread-safe?

Single mutex protects all shared state (queue, counters, shutdown flag). Condition variables for efficient waiting (no busy-polling). Atomic transitions: lock → modify → signal → unlock.

Q8: What memory issues can arise?

  1. Leak: submit allocates nodes, must free in worker and destroy. 2. Use-after-free: don’t access pool after destroy. 3. Task argument lifetime: caller ensures arg valid until task executes. Solutions: careful ownership semantics, testing with ASan/TSan.

Q9: How would you add a task priority feature?

Replace FIFO linked list with priority queue (heap). Submit takes priority parameter. Workers always pop highest priority. Trade-off: O(log n) insert/remove vs O(1) for FIFO.

Q10: How would you make this pool dynamically resizable?

Track min/max workers. Monitor queue depth. If queue grows past threshold, spawn new workers (up to max). If idle for too long, let excess workers exit (down to min). Needs careful synchronization for thread array modification.

Character Device Driver (2 min):

“I built a production character device driver that creates /dev/chardevN nodes. The architecture follows VFS → driver via file_operations. For multi-device support, I embed struct cdev in my device struct and use container_of() in open() to recover per-device state. Read and write use copy_to/from_user for safe kernel-user transfer, with mutex_lock_interruptible for concurrency. I added IOCTL commands for device control using _IO/_IOR/_IOW macros, and sysfs attributes for runtime monitoring without needing a userspace tool. Error handling follows the goto cleanup pattern throughout.”

I2C Temperature Sensor Driver (2 min):

“I wrote a Linux I2C driver for TMP102/TMP112 temperature sensors. The driver uses device tree matching — the DT node’s compatible string triggers my probe() function. I use of_device_get_match_data() to get variant-specific configuration (12-bit vs 13-bit resolution) without if/else chains. Temperature reading uses i2c_smbus_read_word_swapped with 1-second caching to prevent bus flooding. I integrated with the hwmon subsystem using devm_hwmon_device_register_with_info(), giving standard sensor tools (lm-sensors) access to temperature data through /sys/class/hwmon.”

Thread Pool (2 min):

“I implemented a production thread pool in C using the producer-consumer pattern. N worker threads wait on a condition variable for tasks. The main thread submits tasks to a mutex-protected queue and signals workers. Tasks execute outside the lock for parallelism. Shutdown is graceful: set flag, broadcast to wake all workers, join threads. I use two condition variables: not_empty for workers waiting for tasks, all_done for thread_pool_wait() blocking until completion. The implementation handles edge cases like spurious wakeups (while loops), empty queue on shutdown, and proper memory cleanup.”
