🧠 THEORY: I2C Transaction Format

WRITE transaction (Master → Slave):
START ──→ ADDR(7) + W(0) ──→ ACK ──→ REG(8) ──→ ACK ──→ DATA(8) ──→ ACK ──→ STOP
 │           │         │      │        │         │        │         │
 SDA↓        Slave     R/W=0  Slave    Register  Slave    Data      Slave
 while       address   =write ACKs     address   ACKs     byte      ACKs
 SCL↑

READ transaction (Master ← Slave):
START ──→ ADDR+W ──→ ACK ──→ REG ──→ ACK
    ──→ Sr ──→ ADDR+R ──→ ACK ──→ DATA ──→ NACK ──→ STOP
                                     │
                                   Slave drives SDA
                                   Master reads data
Sr = Repeated START (no STOP between write-reg and read-data)

Why repeated start?
  Without: another master could grab bus between our write-reg and read-data
  With:    atomic operation, bus stays locked

🧠 THEORY: Key I2C API Functions

/* == READING from I2C device == */

// Read a single byte from register
int val = i2c_smbus_read_byte_data(client, reg_addr);
// Returns: byte value (0-255) or negative error

// Read 16-bit word from register
int val = i2c_smbus_read_word_data(client, reg_addr);
// Returns: 16-bit value (MSB first from bus) or negative error

// Read 16-bit word, byte-swapped (for little-endian hosts)
int val = i2c_smbus_read_word_swapped(client, reg_addr);
// Returns: 16-bit value with bytes swapped for host endianness

/* == WRITING to I2C device == */

// Write single byte to register
int ret = i2c_smbus_write_byte_data(client, reg_addr, value);

// Write 16-bit word to register
int ret = i2c_smbus_write_word_data(client, reg_addr, value);

// Write 16-bit word, byte-swapped
int ret = i2c_smbus_write_word_swapped(client, reg_addr, value);

/* == BLOCK transfers == */
int ret = i2c_smbus_read_i2c_block_data(client, reg, length, buf);
int ret = i2c_smbus_write_i2c_block_data(client, reg, length, buf);

💻 THE DRIVER — Production Quality

// File: ~/embedded_learning/day_30/i2c_temp/temp_sensor.c
//
// I2C Temperature Sensor Driver (TMP102/TMP112 compatible)
// Features:
//   - Device Tree matching with per-variant configuration
//   - hwmon integration (standard sensor interface)
//   - Temperature caching (avoid bus flooding)
//   - sysfs custom attributes
//   - Managed resources (devm_*)
//
#include <linux/init.h>
#include <linux/module.h>
#include <linux/i2c.h>
#include <linux/hwmon.h>
#include <linux/hwmon-sysfs.h>
#include <linux/slab.h>
#include <linux/mutex.h>
#include <linux/jiffies.h>
#include <linux/of.h>
#include <linux/of_device.h>
#include <linux/err.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("I2C Temperature Sensor Driver (TMP102/TMP112)");
MODULE_VERSION("2.0");

/* ==== REGISTER MAP ==== */
/* TMP102 registers */
#define REG_TEMP        0x00    /* Temperature register (16-bit, read-only) */
#define REG_CONFIG      0x01    /* Configuration register */
#define REG_TLOW        0x02    /* Low temperature threshold */
#define REG_THIGH       0x03    /* High temperature threshold */

/* Configuration register bits */
#define CFG_SD          BIT(0)  /* Shutdown mode */
#define CFG_TM          BIT(1)  /* Thermostat mode */
#define CFG_POL         BIT(2)  /* Polarity */
#define CFG_FQ_MASK     (3<<3)  /* Fault queue */
#define CFG_EM          BIT(4)  /* Extended mode (13-bit) */
#define CFG_OS          BIT(7)  /* One-shot */

/* ==== PER-VARIANT CONFIGURATION ==== */
struct sensor_config {
    const char *name;
    int resolution_bits;    /* 12 or 13 */
    int conversion_time_ms; /* How long conversion takes */
};

static const struct sensor_config tmp102_config = {
    .name = "TMP102",
    .resolution_bits = 12,
    .conversion_time_ms = 26,
};

static const struct sensor_config tmp112_config = {
    .name = "TMP112",
    .resolution_bits = 13,
    .conversion_time_ms = 35,
};

/* ==== PER-DEVICE DATA ==== */
struct temp_sensor_data {
    struct i2c_client *client;
    struct device *hwmon_dev;
    const struct sensor_config *config;
    struct mutex lock;

    /* Temperature cache */
    int cached_temp;            /* milli-degrees Celsius */
    unsigned long last_update;  /* jiffies timestamp */
    bool cache_valid;
};

/* ==== CORE TEMPERATURE READING ==== */

/* Convert raw register value to milli-Celsius */
static int raw_to_millicelsius(int raw, int resolution_bits)
{
    /*
     * TMP102 (12-bit): Bits [15:4] = temperature, [3:0] = 0
     *   Raw value already shifted left by 4
     *   Each LSB = 0.0625°C = 62.5 m°C
     *   Formula: (raw >> 4) * 625 / 10
     *
     * TMP112 (13-bit): Bits [15:3] = temperature, [2:0] = 0
     *   Raw value shifted left by 3
     *   Each LSB = 0.0625°C
     *   Formula: (raw >> 3) * 625 / 10
     */
    int shift = 16 - resolution_bits;
    int temp;

    /* Sign extend if negative (two's complement) */
    if (raw & 0x8000)
        raw |= 0xFFFF0000;  /* Sign extend to 32-bit */

    temp = (raw >> shift) * 625 / 10;  /* Result in milli-Celsius */
    return temp;
}

/* Read temperature with caching */
static int read_temperature(struct temp_sensor_data *data)
{
    int ret;

    mutex_lock(&data->lock);

    /* Check cache validity (1 second) */
    if (data->cache_valid &&
        time_before(jiffies, data->last_update + HZ)) {
        ret = data->cached_temp;
        goto out;
    }

    /* Read temperature register (16-bit, MSB first) */
    ret = i2c_smbus_read_word_swapped(data->client, REG_TEMP);
    if (ret < 0) {
        dev_err(&data->client->dev,
                "Failed to read temperature: %d\n", ret);
        goto out;
    }

    /* Convert raw to milli-Celsius */
    data->cached_temp = raw_to_millicelsius(ret,
                                             data->config->resolution_bits);
    data->last_update = jiffies;
    data->cache_valid = true;
    ret = data->cached_temp;

    dev_dbg(&data->client->dev, "Temperature: %d.%03d°C (raw=0x%04x)\n",
            ret / 1000, abs(ret % 1000), (unsigned int)(ret & 0xFFFF));

out:
    mutex_unlock(&data->lock);
    return ret;
}

/* ==== HWMON INTERFACE ==== */

/* hwmon provides standard sysfs: /sys/class/hwmon/hwmonN/temp1_input */
static int temp_sensor_read(struct device *dev, enum hwmon_sensor_types type,
                            u32 attr, int channel, long *val)
{
    struct temp_sensor_data *data = dev_get_drvdata(dev);
    int ret;

    if (type != hwmon_temp)
        return -EOPNOTSUPP;

    switch (attr) {
    case hwmon_temp_input:
        ret = read_temperature(data);
        if (ret < -500000)  /* Sensor error, not just cold temp */
            return ret;
        *val = ret;  /* milli-Celsius */
        return 0;

    default:
        return -EOPNOTSUPP;
    }
}

static umode_t temp_sensor_is_visible(const void *data,
                                       enum hwmon_sensor_types type,
                                       u32 attr, int channel)
{
    if (type == hwmon_temp && attr == hwmon_temp_input)
        return 0444;  /* Read-only */
    return 0;
}

static const struct hwmon_channel_info * const temp_sensor_info[] = {
    HWMON_CHANNEL_INFO(temp, HWMON_T_INPUT),
    NULL
};

static const struct hwmon_ops temp_sensor_ops = {
    .is_visible = temp_sensor_is_visible,
    .read = temp_sensor_read,
};

static const struct hwmon_chip_info temp_sensor_chip_info = {
    .ops = &temp_sensor_ops,
    .info = temp_sensor_info,
};

/* ==== CUSTOM SYSFS ATTRIBUTES ==== */

static ssize_t sensor_name_show(struct device *dev,
                                 struct device_attribute *attr, char *buf)
{
    struct i2c_client *client = to_i2c_client(dev);
    struct temp_sensor_data *data = i2c_get_clientdata(client);
    return scnprintf(buf, PAGE_SIZE, "%s\n", data->config->name);
}
static DEVICE_ATTR_RO(sensor_name);

static ssize_t resolution_show(struct device *dev,
                                struct device_attribute *attr, char *buf)
{
    struct i2c_client *client = to_i2c_client(dev);
    struct temp_sensor_data *data = i2c_get_clientdata(client);
    return scnprintf(buf, PAGE_SIZE, "%d\n", data->config->resolution_bits);
}
static DEVICE_ATTR_RO(resolution);

static ssize_t raw_temp_show(struct device *dev,
                              struct device_attribute *attr, char *buf)
{
    struct i2c_client *client = to_i2c_client(dev);
    struct temp_sensor_data *data = i2c_get_clientdata(client);
    int ret;

    ret = i2c_smbus_read_word_swapped(client, REG_TEMP);
    if (ret < 0)
        return ret;

    return scnprintf(buf, PAGE_SIZE, "0x%04x\n", ret);
}
static DEVICE_ATTR_RO(raw_temp);

static struct attribute *temp_sensor_attrs[] = {
    &dev_attr_sensor_name.attr,
    &dev_attr_resolution.attr,
    &dev_attr_raw_temp.attr,
    NULL,
};

static const struct attribute_group temp_sensor_group = {
    .attrs = temp_sensor_attrs,
};

static const struct attribute_group *temp_sensor_groups[] = {
    &temp_sensor_group,
    NULL,
};

/* ==== PROBE & REMOVE ==== */

static int temp_sensor_probe(struct i2c_client *client)
{
    struct temp_sensor_data *data;
    struct device *dev = &client->dev;
    int ret;

    dev_info(dev, "Probing I2C temp sensor at 0x%02x\n", client->addr);

    /* Allocate driver data (managed) */
    data = devm_kzalloc(dev, sizeof(*data), GFP_KERNEL);
    if (!data)
        return -ENOMEM;

    data->client = client;
    mutex_init(&data->lock);
    i2c_set_clientdata(client, data);

    /* Get variant config from DT match */
    data->config = of_device_get_match_data(dev);
    if (!data->config) {
        /* Fallback for non-DT matching */
        data->config = &tmp102_config;
    }

    dev_info(dev, "Sensor variant: %s (%d-bit resolution)\n",
             data->config->name, data->config->resolution_bits);

    /* Verify sensor is responsive */
    ret = i2c_smbus_read_word_swapped(client, REG_TEMP);
    if (ret < 0) {
        dev_err(dev, "Failed to read sensor: %d\n", ret);
        return ret;
    }
    dev_info(dev, "Initial temp raw: 0x%04x\n", ret);

    /* Configure sensor */
    ret = i2c_smbus_read_word_swapped(client, REG_CONFIG);
    if (ret < 0) {
        dev_err(dev, "Failed to read config: %d\n", ret);
        return ret;
    }
    dev_info(dev, "Config register: 0x%04x\n", ret);

    /* Create custom sysfs attributes */
    ret = sysfs_create_groups(&client->dev.kobj, temp_sensor_groups);
    if (ret)
        dev_warn(dev, "Failed to create sysfs attrs: %d\n", ret);

    /* Register with hwmon subsystem */
    data->hwmon_dev = devm_hwmon_device_register_with_info(
        dev, "temp_sensor", data,
        &temp_sensor_chip_info, NULL);
    if (IS_ERR(data->hwmon_dev)) {
        ret = PTR_ERR(data->hwmon_dev);
        dev_err(dev, "Failed to register hwmon: %d\n", ret);
        sysfs_remove_groups(&client->dev.kobj, temp_sensor_groups);
        return ret;
    }

    /* Read first temperature */
    int temp = read_temperature(data);
    if (temp > -500000)
        dev_info(dev, "Current temperature: %d.%03d°C\n",
                 temp / 1000, abs(temp % 1000));

    dev_info(dev, "Probe complete (%s, %d-bit)\n",
             data->config->name, data->config->resolution_bits);
    return 0;
}

static void temp_sensor_remove(struct i2c_client *client)
{
    struct temp_sensor_data *data = i2c_get_clientdata(client);

    sysfs_remove_groups(&client->dev.kobj, temp_sensor_groups);

    dev_info(&client->dev, "Removed (%s)\n", data->config->name);
}

/* ==== DEVICE MATCHING ==== */

/* Device Tree match table */
static const struct of_device_id temp_sensor_of_match[] = {
    { .compatible = "ti,tmp102",         .data = &tmp102_config },
    { .compatible = "ti,tmp112",         .data = &tmp112_config },
    { .compatible = "myvendor,temp-sensor", .data = &tmp102_config },
    { }
};
MODULE_DEVICE_TABLE(of, temp_sensor_of_match);

/* Static I2C device ID table (for non-DT systems) */
static const struct i2c_device_id temp_sensor_id[] = {
    { "tmp102", 0 },
    { "tmp112", 0 },
    { "temp-sensor", 0 },
    { }
};
MODULE_DEVICE_TABLE(i2c, temp_sensor_id);

/* I2C driver structure */
static struct i2c_driver temp_sensor_driver = {
    .driver = {
        .name = "temp_sensor",
        .of_match_table = temp_sensor_of_match,
    },
    .probe = temp_sensor_probe,
    .remove = temp_sensor_remove,
    .id_table = temp_sensor_id,
};

/* module_i2c_driver creates init/exit for I2C driver registration */
module_i2c_driver(temp_sensor_driver);

📝 INTERVIEW QUESTIONS — I2C Driver (12 Questions)

Q1: Walk me through your I2C driver’s probe path.

Kernel matches DT compatible “ti,tmp102” to my of_match_table. probe() called with i2c_client. I use devm_kzalloc for data, of_device_get_match_data for config. Verify sensor responds with i2c_smbus_read. Register with hwmon via devm_hwmon_device_register_with_info. Read initial temp.

Q2: Why use i2c_smbus_read_word_swapped vs i2c_smbus_read_word_data?

I2C sends MSB first. On little-endian hosts (x86, ARM-LE), the bytes arrive reversed. _swapped corrects byte order automatically. Without it, you’d need manual be16_to_cpu().

Q3: Why cache temperature readings?

Prevents I2C bus flooding if userspace polls rapidly (cat in a loop). Sensor has ~30ms conversion time anyway — faster reads return stale data. Cache valid for 1 second (HZ jiffies). Protected by mutex for thread safety.

Q4: What is the hwmon subsystem?

Standard Linux framework for hardware monitoring. Provides consistent sysfs interface (/sys/class/hwmon/). Works with sensors command. Temperature in milli-degrees Celsius. My driver just provides read callback — hwmon handles sysfs creation.

Q5: How does device tree matching work in your driver?

DT node has compatible = “ti,tmp102”. My driver’s of_match_table lists this string with .data pointing to tmp102_config struct. Kernel finds match, calls probe. I get config via of_device_get_match_data — no string comparison needed!

Q6: How would you add support for a new sensor variant?

  1. Add new sensor_config struct with resolution/timing. 2. Add compatible string + config pointer to of_match_table. 3. Add to i2c_device_id table. Done — probe() already handles variants via config struct.

Q7: What is i2c_set_clientdata/i2c_get_clientdata?

Stores private data pointer in i2c_client struct. Set in probe, get in other callbacks. Like file->private_data but for I2C. Allows recovery of driver state.

Q8: What errors can I2C operations return?

-ENODEV: no ACK (device not present). -ETIMEDOUT: bus stuck (clock stretching too long). -EIO: general I/O error (bus error). -EBUSY: bus arbitration lost. Always check return values!

Q9: What is the difference between i2c_driver and platform_driver?

i2c_driver: for devices on I2C bus, receives i2c_client. platform_driver: for SoC peripherals, receives platform_device with MMIO resources. Different bus types, same probe/remove model.

Q10: How is MODULE_DEVICE_TABLE used?

Creates alias information put in module ELF. depmod reads these aliases. When kernel detects new I2C device (from DT), it uses these aliases to find and auto-load the correct module.

Q11: What is devm_hwmon_device_register_with_info?

Registers sensor with hwmon subsystem using ops/info structs. hwmon creates sysfs files (temp1_input, etc.) and calls your .read()/.write() callbacks. devm_: auto-unregisters on remove.

Q12: How would you add alarm thresholds to your sensor?

  1. Add HWMON_T_MAX and HWMON_T_MIN to channel info. 2. In read callback, handle hwmon_temp_max and hwmon_temp_min. 3. In write callback (make_visible returns 0644), write thresholds to REG_THIGH/REG_TLOW. 4. Optionally: request IRQ for alert pin, report alarm via hwmon.
